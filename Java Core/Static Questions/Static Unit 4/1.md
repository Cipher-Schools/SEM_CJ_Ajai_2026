class DaysCalculatorImpl {
    public static DaysCalculator createCalculator() {
        return new DaysCalculator() {
            public int calculateDays(int[] date1, int[] date2) {
              
                if (date1[0] < 2000 || date1[0] > 2024 || 
                    date2[0] < 2000 || date2[0] > 2024) {
                    throw new IllegalArgumentException("Year must be between 2000 and 2024");
                }
              
                int totalDays = 0;
                int[] startDate, endDate;
              
                if (date1[0] < date2[0] || (date1[0] == date2[0] && date1[1] < date2[1]) || 
                    (date1[0] == date2[0] && date1[1] == date2[1] && date1[2] <= date2[2])) {
                    startDate = date1;
                    endDate = date2;
                } else {
                    startDate = date2;
                    endDate = date1;
                }
              
                int[] daysInMonth = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
              
                while (startDate[0] < endDate[0] || startDate[1] < endDate[1] || startDate[2] < endDate[2]) {
                    totalDays++;
                  
                    startDate[2]++;
                  
                    int maxDays = daysInMonth[startDate[1]];
                    if (startDate[1] == 2 && isLeapYear(startDate[0])) {
                        maxDays = 29;
                    }
                  
                    if (startDate[2] > maxDays) {
                        startDate[2] = 1;
                        startDate[1]++;
                      
                        if (startDate[1] > 12) {
                            startDate[1] = 1;
                            startDate[0]++;
                        }
                    }
                }
              
                return totalDays;
            }
          
            private boolean isLeapYear(int year) {
                return year % 4 == 0 && (year % 100 != 0 || year % 400 == 0);
            }
        };
    }
}
